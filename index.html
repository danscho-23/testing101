// tiny IOC matcher
targets = ['login', 'verify', 'secure', 'update'];
function suspicious(url){
  const u = new URL(url);
  return targets.some(t => u.hostname.includes(t)) && !u.hostname.endsWith('.trustedcorp.com');
}
